let rec rev_map f ys xs =
  match xs with
  | [] ->
      ys
  | x :: xs ->
      let y = f x in
      rev_map f (y :: ys) xs
      
let map f xs =
  let ys = rev_map f [] xs in
  rev ys
\end{OCaml}
