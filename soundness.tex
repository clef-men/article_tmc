    \begin{align*}
            &
            \iPersistent \left(
                \forall \Psi, e_s, e_t \ldotp
                \Chi (\Psi, e_s, e_t) \iSepImp
                \mathrm{sim \mathhyphen inner}_\bot (\lambdaAbs (\_, e_s, e_t) \ldotp \iSim[\Chi]{\Psi}{e_s}{e_t}) (\bot, e_s, e_t)

            &
            \iSim[\Chi]{\Phi}{e_s}{e_t} \iSepImp

\end{theorem}

\end{theorem}
